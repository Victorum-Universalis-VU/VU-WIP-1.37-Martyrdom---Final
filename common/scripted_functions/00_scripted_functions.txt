		}
	}
	
	### Non-colonial Subjects can't colonize other continents
	condition = {
		tooltip = "VU_Subject_Colonization"
		
		potential = {
			FROM = { is_colonial_nation = no }
			FROM = { is_subject = yes }
			FROM = { is_lesser_in_union = no } #Personal Unions can colonize other continents
		}
		
		allow = {
			FROM = { capital_scope = { continent = ROOT } }
		}
	}
	
	### Colonial Nations can't use Siberian Frontier (added for balance)
	condition = {					
		tooltip = "VU_No_Frontier_Colonization"
		
		potential = {
			FROM = { is_colonial_nation = yes }
			NOT = { FROM = { is_subject_of_type = self_governing_dominion_vu } }
		}
		
		allow = {
			always = no
		}
	}
	
	### Historical Colonisation
	
	#Russian Only
	condition = {
		tooltip = "VU_Historical_Colonization"
		
        potential = {
			NOT = {				
				OR = {
					#Russian Cultures
				    FROM = { primary_culture = byelorussian } #Belarusian
				    FROM = { primary_culture = cossack } #Cossack
				    FROM = { primary_culture = novgorodian } #Novgorodian
				    FROM = { primary_culture = russian } #Russian
				    FROM = { primary_culture = rusyn } #Russyn
				    FROM = { primary_culture = ruthenian } #Ruthenian
				}
			}
			
			FROM = { ai = yes }
			
			OR = {
				#Siberia
				region = ural_region
				region = west_siberia_region
				region = east_siberia_region
				
				#New places that were colonized ONLY by Russia go here
			}
		}
		
		allow = {
			OR = {
				has_global_flag = vu_no_historical_colonization
				FROM = { 
					OR = {
						technology_group = north_american
						technology_group = mesoamerican
						technology_group = south_american
						technology_group = andean
						technology_group = high_american
						technology_group = east_african
						technology_group = central_african
						technology_group = sub_saharan
						technology_group = aboriginal_tech
						technology_group = polynesian_tech
						NOT = { FROM = { is_subject_of_type = self_governing_dominion_vu } }
					}
				}
			}
		}
    }
