country_event = { #The Year of Blood 
	id = VU_Dith_Events.2
	title = VU_Dith_Events.2.t 
	desc = VU_Dith_Events.2.d 
	picture = REVOLUTION_eventPicture
	
	fire_only_once = yes 
	major = yes 
	
	trigger = {
		tag = DTT 
		has_global_flag = great_peasants_war_flag
		is_at_war = no 
		is_bankrupt = no
		total_development = 75
		has_reform = peasants_republic
	}
	
	mean_time_to_happen = { 
		months = 42
	}
	
	option = {
		name = VU_Dith_Events.2.a 
		
		custom_tooltip = dtt_monarch_coalition
		hidden_effect = {
			every_country = {
				limit = {
					capital_scope = {
						region = north_german_region 
						is_part_of_hre = yes 
					}
					NOT = { alliance_with = ROOT } 
					NOT = { 	
						has_opinion = {
							who = ROOT 
							value = 150 
						}
					}
					NOT = { is_subject_of = ROOT }
					OR = {
						is_rival = ROOT 
						government = monarchy 
					}
				}
				form_coalition_against = ROOT
				add_opinion = {
					who = ROOT 
					modifier = dtt_open_peasant_revolts_opinion
				}
				set_ai_attitude = {
					attitude = attitude_hostile
					who = ROOT 
				}
			}
		}
		hidden_effect = {
			if = {
				limit = {
					emperor = {
						is_in_coalition = yes
						coalition_target = ROOT
					}
				}
				emperor = {
					declare_war_with_cb = {
						who = ROOT
						casus_belli = cb_super_badboy
					}
				}
			}
			else = {
				random_country = {
					limit = {
						is_in_coalition = yes
						coalition_target = ROOT
					}
					declare_war_with_cb = {
						who = ROOT
						casus_belli = cb_super_badboy
					}
					save_global_event_target_as = root_coalition_member_peasant_event_target
				}
			}
		}
		add_country_modifier = {
			name = dtt_open_peasant_revolts_modifier
			duration = 1825
		}
		set_country_flag = dtt_openrevolt_flag
		custom_tooltip = dtt_peasants_war_outcome_tt
	}
}
