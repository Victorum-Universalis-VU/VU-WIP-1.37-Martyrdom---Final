	#Italian Roman
	option = {
		name = vu_restore_roman_culture.EVTOPT2

		trigger = {
			NOT = { has_country_flag  = vu_restored_italian_culture }
			num_of_owned_provinces_with = {
				value = 25
				culture_group = latin
				region = italy_region
			}
		}

		custom_tooltip = vu_change_italian_to_roman
		hidden_effect = {
			change_variable = {
		        	which = vu_romano_culture_to_convert
		        	value = 1
	   		}

			every_province = {
				limit = {
					owned_by = ROOT
					culture_group = latin
					region = italy_region
				}
				change_culture = roman
			}
		}
		set_country_flag = vu_restored_italian_culture
		clr_country_flag = used_vu_restore_roman_culture_decision

		ai_chance = { factor = 100 }
	}
