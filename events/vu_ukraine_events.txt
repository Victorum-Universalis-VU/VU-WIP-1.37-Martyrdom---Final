namespace = vu_ukraine
#Tsardom or cossacks
country_event = {
	id = vu_ukraine.1
	title = "vu_ukraine_tsardom_title"
	desc = {
		trigger = {
			has_reform = cossacks_reform
		}
		desc = "vu_ukraine_tsardom_s_desc"
	}
	desc = {
		trigger = {
			NOT = {
				has_reform = cossacks_reform
			}
		}
		desc = "vu_ukraine_tsardom_t_desc"
	}
	picture = GOOD_WITH_MONARCH_eventPicture
	is_triggered_only = yes
	fire_only_once = yes
	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = "vu_ukraine_tsardom_tsar"
		if = {
			limit = {
				is_revolutionary = no
				NOT = {
					government = monarchy
				}
			}
			change_government_to_monarchy = yes
		}
		if = {
			limit = {
				NOT = {
					has_reform = ruthenian_tsardom
				}
			}
			add_government_reform = ruthenian_tsardom
		}
		if = {
			limit = {
				has_custom_ideas = no
				NOT = {
					has_idea_group = ruthenian_ideas
				}
			}
			country_event = {
				id = ideagroups.1
			}
			#Swap Ideas
		}
	}
	option = {
		name = "vu_ukraine_tsardom_hetman"
		set_government_rank = 3
		if = {
			limit = {
				is_revolutionary = no
				NOT = {
					government = republic
				}
			}
			change_government_to_republic = yes
		}
		if = {
			limit = {
				NOT = {
					has_reform = cossacks_reform
				}
			}
			add_government_reform = cossacks_reform
		}
		if = {
			limit = {
				has_custom_ideas = no
				NOT = {
					has_idea_group = cossack_ideas
				}
			}
			country_event = {
				id = ideagroups.1
			}
			#Swap Ideas
		}
	}
}
