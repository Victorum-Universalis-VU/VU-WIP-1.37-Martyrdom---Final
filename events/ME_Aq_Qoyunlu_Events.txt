country_event = {
	id = ME_Aq_Qoyunlu_Events.1 #Uzun Hasan seizes the throne
	title = ME_Aq_Qoyunlu_Events.1.t
	desc = ME_Aq_Qoyunlu_Events.1.d
	picture = COUNTRY_COLLAPSE_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = AKK
		has_global_flag = death_of_shah_rukh
	}
	
	major = yes
	major_trigger = {
		OR = {
			capital_scope = {
				region = mashriq_region
				region = arabia_region
				region = persia_region
				region = egypt_region
				region = anatolia_region
				region = balkan_region
				region = khorasan_region
				region = caucasia_region
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = ME_Aq_Qoyunlu_Events.1.a
		
		ai_chance = { factor = 100 }
		
		if ={ #Uzun Hasan shouldn't die if he already is the ruler.
			limit = {
				NOT = { has_ruler = "Uzun Hasan" }
			}
			kill_ruler = yes
		}	
		if = {
			limit = {
				ai = no
			}
			add_ruler_modifier = {
				name = AKK_rise_of_uzun_hasan
				duration = -1
			}
		}
		else = {
			add_ruler_modifier = {
				name = AKK_rise_of_uzun_hasan_ai
				duration = -1
			}
			capital_scope = {
				infantry = ROOT
				infantry = ROOT
				infantry = ROOT
				infantry = ROOT
				infantry = ROOT
			}
			define_ruler_to_general = {
				fire = 4	
				shock = 5	
				manuever = 5	
				siege = 0
			}
		}
		
		hidden_effect = {
			set_ai_personality = {
				personality = ai_militarist
				locked = yes
			}
		}
	}
}
