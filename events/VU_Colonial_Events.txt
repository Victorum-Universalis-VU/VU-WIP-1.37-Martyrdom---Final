#Colonial Land Transfer
country_event = {
	id = VU_COL_Events.01
	title = VU_COL_Events.01.t
	desc = VU_COL_Events.01.d
	picture = COURT_eventPicture
	
	trigger = {
		FROM = { is_colonial_nation = yes }
		any_owned_province = {
			NOT = { colonial_region = capital }
	
		}
	}

	mean_time_to_happen = { months = 6 }

	#Transfer
	option = {
		name = VU_COL_Events.01.a
		
		hidden_effect = {
			capital_scope = {
				save_event_target_as = colonial_capital
			}
			overlord = {
				save_event_target_as = overlord_country
			}
			every_owned_province = {
				limit = {
					NOT = { colonial_region = event_target:colonial_capital }
				}
				remove_core = ROOT
				add_core = event_target:overlord_country
				cede_province = event_target:overlord_country
			}
		}
	}
}

# No Hard to Colonise Claims
country_event = {
	id = VU_COL_Events.04
	title = VU_COL_Events.04.t
	desc = VU_COL_Events.04.d
	picture = COURT_eventPicture
	
	hidden = yes
	
	trigger = {
	    # If player selected to disable 'Hard to Colonise' this event will NEVER fire.
		NOT = { has_global_flag = malaria_turned_off }
		
		# Diplomatic Technology 47 (~1850s) will unlock malaria provinces, preventing this event.
		FROM = { NOT = { dip_tech = 47 } }
		
		# If the province isn't from the same continent as the owner's capital, the event will trigger.
		# Colonial nations should still be affected - at least before their independence.
		any_province = {
			has_province_modifier = vu_malaria_present
			is_claim = ROOT
			OR = {
				NOT = { same_continent = ROOT }
				AND = { 
					FROM = { is_colonial_nation = yes }
					FROM = { NOT = { is_subject_of_type = self_governing_dominion_vu } }
				}
			}
		}
	}

	mean_time_to_happen = { months = 6 }

	# Remove
	option = {
		name = VU_COL_Events.04.a
		
		hidden_effect = {
			every_claimed_province = {
				limit = {
					has_province_modifier = vu_malaria_present
					OR = {
						NOT = { same_continent = ROOT }
						FROM = { is_colonial_nation = yes }
					}
				}
				remove_claim = ROOT
			}
		}
	}
}
