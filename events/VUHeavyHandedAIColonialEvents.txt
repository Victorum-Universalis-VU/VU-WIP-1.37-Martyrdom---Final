# Heavy handed events to force colonisation for the AI's (now enabled) missions.
#Goes in order of -
#1) Caribbean			#2) America 			#3) Canada 				#4) Newfoundland	#5) Hudson Bay
#6) Greenland			#7) Acadia				#8) Cuba					#9) Hispaniola		#10) Bahamas
#11) NWTs			#12) Cascadia			#13) Alaska			#14) Louisiana		#15) California
#16) Mexico			#17) Colombia		#18) Peru				#19) La Plata			#20) Brazil
#21) Australia			#22) Polynesia

#Note Florida is skipped, as it is likely to be cut soon


province_event = {
	id = VictorumUniversalisHeavyHandedAIColonialEvents.1
	title = "VictorumUniversalisHeavyHandedAIColonialEvents.EVTNAME1"
	desc = "VictorumUniversalisHeavyHandedAIColonialEvents.EVTDESC1"
	picture = COLONIZATION_eventPicture

	is_triggered_only = yes

	trigger = {
		owner = { ai = yes }
		any_province = {
			colonial_region = colonial_the_carribean
			vu_historical_colony_events_for_the_ai_candidate = yes
		}
	}

	option = { # Success
		name = "VictorumUniversalisHeavyHandedAIColonialEvents.EVTOPTA1"
		ai_chance = { factor = 99 }
		random_province = {
			limit = {
				colonial_region = colonial_the_carribean
				vu_historical_colony_events_for_the_ai_candidate = yes
			}
			create_colony = 60
			change_native_hostileness = -9
		}
	}

	option = { # Fails
		name = "VictorumUniversalisHeavyHandedAIColonialEvents.EVTOPTB1"
		ai_chance = {
			factor = 1
			modifier = {
				factor = 100
				vu_more_colonies_than_colonists_check = yes
			}
			modifier = {
				factor = 100
				OR = {
					owner = { is_in_deficit = yes }
					owner = { is_bankrupt = yes }
				}
			}
		}
		owner = { add_prestige = -1 }
	}
}
