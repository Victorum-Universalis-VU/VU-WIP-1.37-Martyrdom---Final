country_event = { # Muscovy stops paying tribute to Sarai
	id = vu_GOL.1
	title = vu_GOL.1.t
	desc = vu_GOL.1.d
	picture = LAND_MILITARY_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	goto = 6975
	
	trigger = {
		tag = GOL
		is_year = 1469
		exists = MOS
		NOT = {
			MOS = {
				has_opinion = {
					who = ROOT
					value = 25
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		add_casus_belli = {
			target = MOS
			type = cb_force_tributary
			months = 12
		}
	}
	
	option = { # We will put them back under yoke
		name = vu_GOL.1.a
		highlight = yes
		ai_chance = { factor = 100 } #AI will always launch punitive raid, to accelerate the downfall and chance of further events
		declare_war_with_cb = {
			who = MOS
			casus_belli = cb_force_tributary
		}
		set_country_flag = moscov_no_tribute
	}
	
	option = { # We aren't ready for another invasion
		name = vu_GOL.1.b
		ai_chance = { factor = 0 }
		add_prestige = -20
		add_horde_unity = -25
		add_mil_power = -20
		set_country_flag = moscov_no_tribute
	}
}
